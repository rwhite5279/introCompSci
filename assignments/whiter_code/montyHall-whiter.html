# montyhall.py
"""Monty Hall demonstration program, by Richard White, 2010-03-11.
This program is designed to demonstrate the Monty Hall problem, or at least allow the user to experiment with it.

The "Monty Hall" problem is this:
In the TV game show "Let's Make a Deal," master of ceremonies Monty Hall invites contestants to choose door #1, #2, or #3, behind one of which is a prize car. Behind the other two doors is typically a goat. After the contestant picks a door (say Door #1 for our example), Monty Hall will open one of the other doors (say Door #2) to reveal one of the goats. "Now that you've seen that, do you want to change your mind? You can still pick Door #3 if you choose!"

Does the contestant have a better chance of winning the car if he stays with his original choice (#1), switches to the new door (#3), or are the odds of winning the car equal between #1 and #3?

This version of the program is simply a text version that allows the user to choose the doors, and then reveals the results. A logical extension of the current program would be to make a graphics version, more appealing to the user. It would also be nice to run some actual statistics on this, to explain the statistical analysis, and then give some real results of 100 trials, say, so that they can see some evidence of the analysis."""

# Reveal one of the other doors

# Reveal answer
# Play again?


















































# ------------------ PROGRAM ---------------------------

import random

def intro():
    print "We're going to play \"Let's Make a Deal!\""
    print "There's a car hidden behind Door 1, 2, or 3."
    print "Behind the other doors are goats."
    print "Let's see if you can win that car!!!"

def play():  
    # initialize doors with random prizes
    doors = []
    prizes = ["car","goat","goat"]  # These are the prizes to choose from

    # Go through list of prizes, and pop them off randomly, putting them
    # behind doors.
    while len(prizes) > 0:          
        doors.append(prizes.pop(random.randint(0,len(prizes)-1)))

    # Now we can print out the doors for people
    for i in range(1,4):
        print "[  %1d  ]" % i,
    print
    doorA = input("Which door which you like to choose (1, 2, or 3)?")
    print "You've chosen Door #",doorA
    find_a_goat = doorA-1               # Sets up variable so that we run through
                                        # loop to find the other goat door
                                        # Note that when doorA = 1, find_a_goat = 0
    while ((find_a_goat == doorA-1) or (doors[find_a_goat] == "car")):
        find_a_goat = random.randint(0,2)
    for i in range(1,4):                # Find last door and assign it to doorB
        if ((i != doorA) and (i != find_a_goat+1)):
            doorB = i
            break
    print "Now before we look behind that door, let's"
    print "take a look at what's behind door #",find_a_goat+1
    for i in range(1,4):                # Go through doors
        if (i == find_a_goat+1):        # If it's the goat one...
            print "[",doors[i-1],"]",   # ... print it out
        else:
            print "[  %1d ]" % i,       # Otherwise, just print the number
    print
    print "It's a good thing you didn't choose that one!"
    print "Now, let me ask you: you've selected Door #",doorA
    print "Do you want to KEEP that door, or do you want to switch from"
    print "Door #",doorA,"to Door #",doorB,"?"
    print "(Enter",doorA,"or",doorB,") ",
    final_choice = input()
    print "You chose Door #",final_choice,", and you won a",doors[final_choice-1],"."

# --------------CALL THE FUNCTIONS------------------------

def main():
    intro()
    play_again = "y"
    while (play_again[0].lower() == "y"):
        play()
        play_again = raw_input("Wanna play again? y/n ")
        
if __name__ == "__main__":
    main()
